一.质数

1. 判断一个数是不是质数：

试除法判定质数 时间复杂度 O(n)O(\sqrt{n})O(n​)

代码模板：


bool is_prime(int x){
    if(x<2) return false;
    for(int i=2;i<=x/i;i++){
        if(x%i==0) return false;
    }
    return true;
}

2. 线性筛出111~nnn中的质数：

线性筛 时间复杂度 O(n)O(n)O(n)

代码模板：


void get_prime(int n){
   for(int i=2;i<=n;i++){
        if(!st[i]) prime[cnt++]=i;
       for(int j=0;prime[j]<=n/i;j++){
            st[prime[j]*i]=true;
            if(i%prime[j]==0) break;
        }
    }
}

3. 二次筛法筛出lll~rrr中的质数：

当 lll 和 rrr 的范围大于 1e61e61e6 的时候就可以使用二次筛法来筛出 lll~rrr 的质数

题目链接： 质数距离

二次筛法 时间复杂度 近似O(n)O(n)O(n)

代码模板：


#include<bits/stdc++.h>
using namespace std;
#define int long long
const int N=1e6+10;
bool st[N];
int prime[N];
int cnt=0;

void prime_get(int x){
   
    memset(st,0,sizeof st);
      cnt=0;
    for(int i=2;i<=x;i++){
        if(!st[i]) prime[cnt++]=i;
        for(int j=0;prime[j]<=x/i;j++){
            st[prime[j]*i]=true;
            if(i%prime[j]==0) break;
        }
    }
}

signed main(){
    int l,r;
    while(cin>>l>>r){
        prime_get(50010);
       memset(st,0,sizeof st);

       for(int i=0;i<cnt;i++){
           int p=prime[i];
           for(int j=max(((l+p-1)/p)*p,2*p);j<=r;j+=p){
               st[j-l]=true;
           }
    } 
     cnt=0;
    for(int i=l;i<=r;i++){
        if(!st[i-l]&&i>=2)  prime[cnt++]=i;
    }    
   
    if(cnt<2) cout<<"There are no adjacent primes.\n";
    else {
         int minp=0,maxp=0;
     for(int i=0;i+1<cnt;i++){
         int d=prime[i+1]-prime[i];
         if(d<(prime[minp+1]-prime[minp])) minp=i;
         if(d>(prime[maxp+1]-prime[maxp])) maxp=i;
         
     }
     printf("%lld,%lld are closest, %lld,%lld are most distant.\n",prime[minp],prime[minp+1],prime[maxp],prime[maxp+1]);
  }  
        
    }
    return 0;
}

二. 分解质因数：

1. 分解一个数（范围在 1e121e121e12 以内）的质因数：

时间复杂度 O(n)O(\sqrt{n})O(n​)

代码模板：

void prime_put(int x){
    for(int i=2;i<=x/i;i++){
        if(x%i==0){
            int s=0;
            while(x%i==0){
                x/=i;
                s++;
            }
             cout<<i<<' '<<s<<endl;
        }

    }
    if(x>1) cout<<x<<' '<<1<<endl;
    cout<<endl;
}

2. 阶乘质因数分解：

分解n!n!n! 其中 nnn 的范围是111~1e61e61e6 均可在111秒内跑完,时间复杂度近似nlog(n)nlog(n)nlog(n)

代码模板：

#include<bits/stdc++.h>
using namespace std;
const int N=1e6+10;
int prime[N],cnt;
bool st[N];

//先预处理出所有质数
void prime_get(int n){
    for(int i=2;i<=n;i++){
        if(!st[i]) prime[cnt++]=i;
        for(int j=0;prime[j]<=n/i;j++){
            st[prime[j]*i]=true;
            if(i%prime[j]==0) break;
        }
    }
}
int main(){

    int n;
    cin>>n;
    prime_get(n);
    for(int i=0;i<cnt;i++){
    //取出质因子
        int p=prime[i];
        int s=0;
    //将质因子的所有次方的个数加起来
        for(int j=n;j;j/=p) s+=j/p;
        cout<<p<<' '<<s<<endl;
    }
}
